### if vs guard
```swift 
// if
func greetingWithIf() {
    let value: String? = "lina"
    type(of: value) // Optional<String>
    if let name = value {
        print("hello, \(name)")
        type(of: name) // String
    }

// guard
func greetingWithGuard() {
    let value: String? = "lina"
    type(of: value) // Optional<String>
    guard let name = value else {
        return
    }
    print("hello, \(name)")
    type(of: name) // String
}
```
